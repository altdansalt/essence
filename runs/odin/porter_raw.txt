package main

import "core:fmt"
import "core:strings"
import "core:strconv"

// ---------- Value ----------
VType :: enum u8 { NULL, INT, TEXT }

Value :: struct {
    type: VType,
    ival: i64,
    sval: string,
}

v_null :: proc() Value { return Value{type = .NULL} }
v_int :: proc(i: i64) Value { return Value{type = .INT, ival = i} }
v_text :: proc(s: string) Value { return Value{type = .TEXT, sval = s} }

v_copy :: proc(v: Value) -> Value {
    if v.type == .TEXT { return v_text(v.sval) }
    return v
}

v_cmp :: proc(a, b: Value) -> int {
    if a.type != b.type { return int(a.type) - int(b.type) }
    switch a.type {
    case .INT:
        if a.ival < b.ival { return -1 }
        if a.ival > b.ival { return 1 }
        return 0
    case .TEXT:
        return strings.compare(a.sval, b.sval)
    case .NULL:
        return 0
    }
    return 0
}

v_to_text :: proc(v: Value) -> string {
    switch v.type {
    case .INT: return fmt.tprintf("%d", v.ival)
    case .TEXT: return v.sval
    case .NULL: return "NULL"
    }
    return "NULL"
}

v_truthy :: proc(v: Value) -> bool {
    return v_cmp(v, v_int(0)) != 0
}

lex :: proc(sql: string) -> [dynamic]Token {
    toks: [dynamic]Token
    pos := 0
    src := sql
    push :: proc(toks: ^[dynamic]Token, k: TokKind, s: string) {
        append(toks, Token{kind = k, text = s})
    }
    while pos < len(src) {
        c := src[pos]
        if c == ' ' || c == '\t' || c == '\n' || c == '\r' { pos += 1; continue }
        if c == '-' && pos+1 < len(src) && src[pos+1] == '-' {
            while pos < len(src) && src[pos] != '\n' { pos += 1 }
            continue
        }
        if (c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z') || c == '_' {
            st := pos
            while pos < len(src) && ((src[pos] >= 'a' && src[pos] <= 'z') || (src[pos] >= 'A' && src[pos] <= 'Z') || (src[pos] >= '0' && src[pos] <= '9') || src[pos] == '_') {
                pos += 1
            }
            word := src[st:pos]
            push(&toks, is_kw(word) ? .KW : .ID, word)
            continue
        }
        if c >= '0' && c <= '9' {
            st := pos
            while pos < len(src) && src[pos] >= '0' && src[pos] <= '9' { pos += 1 }
            push(&toks, .INT, src[st:pos])
            continue
        }
        if c == '\'' {
            pos += 1
            st := pos
            while pos < len(src) && src[pos] != '\'' { pos += 1 }
            push(&toks, .STRING, src[st:pos])
            if pos < len(src) && src[pos] == '\'' { pos += 1 }
            continue
        }
        if c == '"' {
            pos += 1
            st := pos
            while pos < len(src) && src[pos] != '"' { pos += 1 }
            push(&toks, .ID, src[st:pos])
            if pos < len(src) && src[pos] == '"' { pos += 1 }
            continue
        }
        switch c {
        case '*': push(&toks, .STAR, "*"); pos += 1
        case ',': push(&toks, .COMMA, ","); pos += 1
        case '(': push(&toks, .LP, "("); pos += 1
        case ')': push(&toks, .RP, ")"); pos += 1
        case ';': push(&toks, .SEMI, ";"); pos += 1
        case '.': push(&toks, .DOT, "."); pos += 1
        case:
            if c == '=' || c == '<' || c == '>' || c == '!' {
                st := pos
                if pos+1 < len(src) && src[pos+1] == '=' { pos += 1 }
                if c == '<' && pos+1 < len(src) && src[pos+1] == '>' { pos += 1 }
                if c == '!' && pos+1 < len(src) && src[pos+1] == '=' { pos += 1 }
                pos += 1
                push(&toks, .OP, src[st:pos])
            } else {
                push(&toks, .PUNCT, string([1]u8{c}))
                pos += 1
            }
        }
    }
    push(&toks, .EOF, "")
    return toks

parse_primary :: proc(p: ^Parser) -> ^Expr {
    t := p.toks[p.i]
    if t.kind == .INT {
        e := new_expr(.INT)
        e.ival = strconv.parse_int(t.text, 10) or_else 0
        p.i += 1
        return e
    }
    if t.kind == .STRING {
        e := new_expr(.STR)
        e.sval = t.text
        p.i += 1
        return e
    }
    if t.kind == .ID || t.kind == .KW {
        e := new_expr(.COL)
        buf := t.text
        p.i += 1
        if p.toks[p.i].kind == .DOT {
            p.i += 1
            buf = fmt.tprintf("%s.%s", buf, p.toks[p.i].text)
            p.i += 1
        }
        e.col = buf
        return e
    }
    if p_accept(p, .LP, "") {
        e := parse_expr(p)
        p_accept(p, .RP, "")
        return e
    }
    return nil

parse_cmp :: proc(p: ^Parser) -> ^Expr {
    l := parse_primary(p)
    if p.toks[p.i].kind == .OP {
        e := new_expr(.BINOP)
        e.op = p.toks[p.i].text
        p.i += 1
        e.l = l
        e.r = parse_primary(p)
        return e
    }
    return l

parse_expr :: proc(p: ^Parser) -> ^Expr {
    l := parse_cmp(p)
    if p_peek_kw(p, "and") || p_peek_kw(p, "or") {
        e := new_expr(.BINOP)
        e.op = p_peek_kw(p, "and") ? "AND" : "OR"
        p.i += 1
        e.l = l
        e.r = parse_expr(p)
        return e
    }

parse_select :: proc(p: ^Parser, st: ^Stmt) -> bool {
    st.is_select = true
    if p_accept(p, .STAR, "") {
        st.sel_star = true
    } else {
        do {
            rc := ResultCol{star = false}
            if p_accept(p, .STAR, "") {
                rc.star = true
            } else if p.toks[p.i].kind == .ID || p.toks[p.i].kind == .KW {
                buf := p.toks[p.i].text
                p.i += 1
                if p.toks[p.i].kind == .DOT {

                    if p_accept(p, .STAR, "") {
                        rc.star = true
                    } else {
                        buf = fmt.tprintf("%s.%s", buf, p.toks[p.i].text)
                        p.i += 1
                        rc.col = buf
                    }
                } else {
                    rc.col = buf
                }
            } else {
                return false
            }
            append(&st.sel_cols, rc)
        } while p_accept(p, .COMMA, "")

    // returns (tabidx, colidx, ok)

    if dot_pos >= 0 {

eval_expr :: proc(e: ^Expr, c: RowCtx) -> Value {
    if e == nil { return v_null() }
    switch e.kind {
    case .INT: return v_int(e.ival)
    case .STR: return v_text(e.sval)
    case .COL:
        ti, ci, ok := rowctx_colindex(c, e.col)
        if !ok { return v_null() }

                cur[k] = 0

VType :: enum u8 { NULL, INT, TEXT }

Value :: struct {
    type: VType,
    ival: i64,
    sval: string,
}

v_null :: proc() Value { return Value{type = .NULL} }
v_int :: proc(i: i64) Value { return Value{type = .INT, ival = i} }

    if a.type != b.type { return int(a.type) - int(b.type) }
    switch a.type {
    case .INT:
        if a.ival < b.ival { return -1 }
        if a.ival > b.ival { return 1 }

lex :: proc(sql: string) -> [dynamic]Token {
    toks: [dynamic]Token
    pos := 0
    src := sql
    for pos < len(src) {
        c := src[pos]
        if c == ' ' || c == '\t' || c == '\n' || c == '\r' { pos += 1; continue }
        if c == '-' && pos+1 < len(src) && src[pos+1] == '-' {
            for pos < len(src) && src[pos] != '\n' { pos += 1 }
            continue
        }
        if (c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z') || c == '_' {
            st := pos
            for pos < len(src) && ((src[pos] >= 'a' && src[pos] <= 'z') || (src[pos] >= 'A' && src[pos] <= 'Z') || (src[pos] >= '0' && src[pos] <= '9') || src[pos] == '_') {
                pos += 1
            }
            word := src[st:pos]
            append(&toks, Token{kind = is_kw(word) ? .KW : .ID, text = word})
            continue
        }
        if c >= '0' && c <= '9' {
            st := pos
            for pos < len(src) && src[pos] >= '0' && src[pos] <= '9' { pos += 1 }
            append(&toks, Token{kind = .INT, text = src[st:pos]})
            continue
        }
        if c == '\'' {
            pos += 1
            st := pos
            for pos < len(src) && src[pos] != '\'' { pos += 1 }
            append(&toks, Token{kind = .STRING, text = src[st:pos]})
            if pos < len(src) && src[pos] == '\'' { pos += 1 }
            continue
        }
        if c == '"' {
            pos += 1
            st := pos
            for pos < len(src) && src[pos] != '"' { pos += 1 }
            append(&toks, Token{kind = .ID, text = src[st:pos]})
            if pos < len(src) && src[pos] == '"' { pos += 1 }
            continue
        }
        switch c {
        case '*': append(&toks, Token{.STAR, "*"}); pos += 1
        case ',': append(&toks, Token{.COMMA, ","}); pos += 1
        case '(': append(&toks, Token{.LP, "("}); pos += 1
        case ')': append(&toks, Token{.RP, ")"}); pos += 1
        case ';': append(&toks, Token{.SEMI, ";"}); pos += 1
        case '.': append(&toks, Token{.DOT, "."}); pos += 1
        case:
            if c == '=' || c == '<' || c == '>' || c == '!' {
                st := pos
                if pos+1 < len(src) && src[pos+1] == '=' { pos += 1 }
                if c == '<' && pos+1 < len(src) && src[pos+1] == '>' { pos += 1 }
                if c == '!' && pos+1 < len(src) && src[pos+1] == '=' { pos += 1 }

                pos += 1
            }
        }

parse_primary :: proc(p: ^Parser) -> ^Expr {
    t := p.toks[p.i]
    if t.kind == .INT {
        e := new_expr(.INT)
        e.ival = strconv.parse_int(t.text, 10) or_else 0
        p.i += 1
        return e
    }
    if t.kind == .STRING {
        e := new_expr(.STR)
        e.sval = t.text
        p.i += 1
        return e
    }
    if t.kind == .ID || t.kind == .KW {
        e := new_expr(.COL)
        buf := t.text
        p.i += 1
        if p.toks[p.i].kind == .DOT {
            p.i += 1
            buf = fmt.tprintf("%s.%s", buf, p.toks[p.i].text)
            p.i += 1
        }
        e.col = buf
        return e
    }
    if p_accept(p, .LP, "") {
        e := parse_expr(p)
        p_accept(p, .RP, "")
        return e
    }

    for p.toks[p.i].kind != .RP && p.toks[p.i].kind != .EOF {
        if p.toks[p.i].kind == .INT {
            append(&st.ins_vals, v_int(strconv.parse_int(p.toks[p.i].text, 10) or_else 0))
            p.i += 1
        } else if p.toks[p.i].kind == .STRING {
            append(&st.ins_vals, v_text(p.toks[p.i].text))
            p.i += 1
        } else if p_accept_kw(p, "null") {
            append(&st.ins_vals, v_null())
        } else {
            return false
        }
        if !p_accept(p, .COMMA, "") { break }
    }
    p_accept(p, .RP, "")
    return true
}

parse_select :: proc(p: ^Parser, st: ^Stmt) -> bool {
    st.is_select = true
    if p_accept(p, .STAR, "") {
        st.sel_star = true
    } else {
        for {
            rc := ResultCol{star = false}
            if p_accept(p, .STAR, "") {
                rc.star = true
            } else if p.toks[p.i].kind == .ID || p.toks[p.i].kind == .KW {
                buf := p.toks[p.i].text
                p.i += 1
                if p.toks[p.i].kind == .DOT {
                    p.i += 1
                    if p_accept(p, .STAR, "") {
                        rc.star = true
                    } else {
                        buf = fmt.tprintf("%s.%s", buf, p.toks[p.i].text)
                        p.i += 1
                        rc.col = buf
                    }
                } else {
                    rc.col = buf
                }
            } else {
                return false
            }
            append(&st.sel_cols, rc)
            if !p_accept(p, .COMMA, "") { break }
        }
    }
    if !p_accept_kw(p, "from") { return false }
    if p.toks[p.i].kind != .ID { return false }
    // first table
    {
        tr := TableRef{tname = p.toks[p.i].text}
        p.i += 1
        if p.toks[p.i].kind == .ID && !p_peek_kw(p, "on") && !p_peek_kw(p, "where") && !p_peek_kw(p, "order") && !p_peek_kw(p, "inner") && !p_peek_kw(p, "join") {
            tr.alias = p.toks[p.i].text
            p.i += 1
        }
        append(&st.sel_tables, tr)
    }
    // joins
    for p_peek_kw(p, "inner") || p_peek_kw(p, "join") {
        if p_accept_kw(p, "inner") {
            if !p_accept_kw(p, "join") { return false }
        } else {
            p_accept_kw(p, "join")
        }
        if p.toks[p.i].kind != .ID { return false }
        tr := TableRef{tname = p.toks[p.i].text}
        p.i += 1
        if p.toks[p.i].kind == .ID && !p_peek_kw(p, "on") && !p_peek_kw(p, "where") && !p_peek_kw(p, "order") {
            tr.alias = p.toks[p.i].text
            p.i += 1
        }
        append(&st.sel_tables, tr)
        if p_accept_kw(p, "on") {
            on := parse_expr(p)
            if st.sel_join_on == nil {
                st.sel_join_on = on
            } else {
                and := new_expr(.BINOP)
                and.op = "AND"
                and.l = st.sel_join_on
                and.r = on
                st.sel_join_on = and
            }
        }
    }
    if p_accept_kw(p, "where") {
        st.sel_where = parse_expr(p)
    }
    if p_accept_kw(p, "order") {
        if !p_accept_kw(p, "by") { return false }
        if p.toks[p.i].kind != .ID && p.toks[p.i].kind != .KW { return false }
        st.sel_order_col = p.toks[p.i].text
        p.i += 1
        if p_accept_kw(p, "desc") {
            st.sel_order_desc = true
        } else {
            p_accept_kw(p, "asc")
        }
    }
    return true
}

parse :: proc(p: ^Parser, st: ^Stmt) -> bool {
    if p_peek_kw(p, "begin") {
        st.is_begin = true
        p.i += 1
        p_accept_kw(p, "transaction")
    } else if p_peek_kw(p, "commit") {
        st.is_commit = true
        p.i += 1
        p_accept_kw(p, "transaction")
    } else if p_accept_kw(p, "create") {
        if p_accept_kw(p, "table") {
            if !parse_create(p, st) { return false }
        } else {
            return false
        }
    } else if p_accept_kw(p, "insert") {
        if !parse_insert(p, st) { return false }
    } else if p_accept_kw(p, "select") {
        if !parse_select(p, st) { return false }
    } else {
        return false
    }
    p_accept(p, .SEMI, "")
    return true
}

// ---------- Database ----------
DB :: struct {
    schema: Schema,
    stores: [dynamic]RowStore,
    errmsg: string,
    in_txn: bool,
}

// ---------- Expression eval ----------
RowCtx :: struct {
    refs: []TableRef,
    tabs: []^Table,
    rows: []Row,
}

rowctx_colindex :: proc(c: RowCtx, col: string) -> (int, int, bool) {
    dot_pos := strings.index(col, ".")
    if dot_pos >= 0 {
        tname := col[:dot_pos]
        cname := col[dot_pos+1:]
        for i in 0..<len(c.refs) {
            tn := c.refs[i].alias != "" ? c.refs[i].alias : c.refs[i].tname
            if strings.equal_fold(tn, tname) {
                ci := table_colindex(c.tabs[i], cname)
                if ci >= 0 { return i, ci, true }
            }
        }
        return 0, 0, false
    }
    for i in 0..<len(c.tabs) {
        ci := table_colindex(c.tabs[i], col)
        if ci >= 0 { return i, ci, true }
    }
    return 0, 0, false
}

eval_expr :: proc(e: ^Expr, c: RowCtx) -> Value {
    if e == nil { return v_null() }
    switch e.kind {
    case .INT: return v_int(e.ival)
    case .STR: return v_text(e.sval)
    case .COL:
        ti, ci, ok := rowctx_colindex(c, e.col)
        if !ok { return v_null() }
        return v_copy(c.rows[ti].vals[ci])
    case .BINOP:
        a := eval_expr(e.l, c)
        b := eval_expr(e.r, c)
        if strings.equal_fold(e.op, "AND") {
            return v_int(v_truthy(a) && v_truthy(b) ? 1 : 0)
        } else if strings.equal_fold(e.op, "OR") {
            return v_int(v_truthy(a) || v_truthy(b) ? 1 : 0)
        }
        cmp := v_cmp(a, b)
        switch e.op {
        case "=": return v_int(cmp == 0 ? 1 : 0)
        case "<>": return v_int(cmp != 0 ? 1 : 0)
        case "<": return v_int(cmp < 0 ? 1 : 0)
        case ">": return v_int(cmp > 0 ? 1 : 0)
        case "<=": return v_int(cmp <= 0 ? 1 : 0)
        case ">=": return v_int(cmp >= 0 ? 1 : 0)
        case: return v_null()
        }
    }
    return v_null()
}

row_match :: proc(where: ^Expr, c: RowCtx) -> bool {
    if where == nil { return true }
    v := eval_expr(where, c)
    return v_truthy(v)
}

// ---------- Prepared statement ----------
Stmt_Handle :: struct {
    db: ^DB,
    st: Stmt,
    started: bool,
    sorted: [dynamic]Row,
    cur_sorted: int,
    outrow: [dynamic]Value,
    nout: int,
}

collect_matching :: proc(s: ^Stmt_Handle) {
    db := s.db
    n := len(s.st.sel_tables)
    idx: [dynamic]int
    for i in 0..<n {
        append(&idx, schema_index(&db.schema, s.st.sel_tables[i].tname))
    }
    tabs: [dynamic]^Table
    for i in 0..<n {
        if idx[i] >= 0 {
            append(&tabs, &db.schema.tables[idx[i]])
        } else {
            append(&tabs, nil)
        }
    }
    cur: [dynamic]int
    for i in 0..<n {
        append(&cur, 0)
    }
    
    for {
        ok := true
        for i in 0..<n {
            if idx[i] < 0 || cur[i] >= len(db.stores[idx[i]].rows) {
                ok = false
                break
            }
        }
        if !ok { break }
        
        rows: [dynamic]Row
        for i in 0..<n {
            append(&rows, db.stores[idx[i]].rows[cur[i]])
        }
        c := RowCtx{refs = s.st.sel_tables[:], tabs = tabs[:], rows = rows[:]}
        
        if row_match(s.st.sel_where, c) && (s.st.sel_join_on == nil || row_match(s.st.sel_join_on, c)) {
            r: Row
            for i in 0..<n {
                for j in 0..<len(rows[i].vals) {
                    append(&r.vals, v_copy(rows[i].vals[j]))
                }
            }
            append(&s.sorted, r)
        }
        
        delete(rows)
        
        // advance innermost
        k := n - 1
        for k >= 0 {
            cur[k] += 1
            if idx[k] >= 0 && cur[k] < len(db.stores[idx[k]].rows) {
                break
            }
            if k > 0 {
                cur[k] = 0
            }
            k -= 1
        }
        if k < 0 { break }
    }
    
    delete(&idx)
    delete(&tabs)
    delete(&cur)
    
    // ORDER BY (single table only)
    if s.st.sel_order_col != "" && len(s.st.sel_tables) == 1 {
        t := schema_find(&db.schema, s.st.sel_tables[0].tname)
        if t != nil {
            cidx := table_colindex(t, s.st.sel_order_col)
            if cidx >= 0 {
                for i in 1..<len(s.sorted) {
                    for j in i; j > 0; j -= 1 {
                        a := s.sorted[j-1].vals[cidx]
                        b := s.sorted[j].vals[cidx]
                        cmp := v_cmp(a, b)
                        if (s.st.sel_order_desc && cmp < 0) || (!s.st.sel_order_desc && cmp > 0) {
                            tmp := s.sorted[j-1]
                            s.sorted[j-1] = s.sorted[j]
                            s.sorted[j] = tmp
                        } else {
                            break
                        }
                    }
                }
            }
        }
    }
}

stmt_step_select :: proc(s: ^Stmt_Handle) -> int {
    if !s.started {
        s.started = true
        collect_matching(s)
    }
    if s.cur_sorted >= len(s.sorted) { return 101 }
    r := s.sorted[s.cur_sorted]
    s.cur_sorted += 1
    n := len(s.st.sel_tables)
    tabs: [dynamic]^Table
    for i in 0..<n {
        append(&tabs, schema_find(&s.db.schema, s.st.sel_tables[i].tname))
    }
    offsets: [dynamic]int
    off := 0
    for i in 0..<n {
        append(&offsets, off)
        off += len(tabs[i].cols)
    }
    clear(&s.outrow)
    if s.st.sel_star {
        for i in 0..<n {
            for j in 0..<len(tabs[i].cols) {
                append(&s.outrow, v_copy(r.vals[offsets[i]+j]))
            }
        }
    } else {
        for k in 0..<len(s.st.sel_cols) {
            rc := s.st.sel_cols[k]
            if rc.star {
                for i in 0..<n {
                    for j in 0..<len(tabs[i].cols) {
                        append(&s.outrow, v_copy(r.vals[offsets[i]+j]))
                    }
                }
            } else {
                dot_pos := strings.index(rc.col, ".")
                ti := 0
                ci := -1
                if dot_pos >= 0 {
                    tname := rc.col[:dot_pos]
                    cname := rc.col[dot_pos+1:]
                    for i in 0..<n {
                        tn := s.st.sel_tables[i].alias != "" ? s.st.sel_tables[i].alias : s.st.sel_tables[i].tname
                        if strings.equal_fold(tn, tname) {
                            ci = table_colindex(tabs[i], cname)
                            ti = i
                            break
                        }
                    }
                } else {
                    for i in 0..<n {
                        ci = table_colindex(tabs[i], rc.col)
                        if ci >= 0 { ti = i; break }
                    }
                }
                if ci < 0 {
                    append(&s.outrow, v_null())
                } else {
                    append(&s.outrow, v_copy(r.vals[offsets[ti]+ci]))
                }
            }
        }
    }
    s.nout = len(s.outrow)
    delete(&tabs)
    delete(&offsets)
    return 100
}

// ---------- Public API ----------
sqlite3_open :: proc(name: string) -> (^DB, int) {
    db := new(DB)
    return db, 0
}

sqlite3_close :: proc(db: ^DB) -> int {
    if db == nil { return 0 }
    for i in 0..<len(db.schema.tables) {
        delete(db.schema.tables[i].cols)
    }
    delete(db.schema.tables)
    for i in 0..<len(db.stores) {
        for j in 0..<len(db.stores[i].rows) {
            delete(db.stores[i].rows[j].vals)
        }
        delete(db.stores[i].rows)
    }
    delete(db.stores)
    free(db)
    return 0
}

sqlite3_errmsg :: proc(db: ^DB) -> string {
    if db == nil { return "" }
    return db.errmsg
}

Exec_Callback :: proc(arg: rawptr, n: int, vals: []string, cols: []string)

sqlite3_exec :: proc(db: ^DB, sql: string, cb: Exec_Callback, arg: rawptr) -> (int, string) {
    toks := lex(sql)
    defer delete(toks)
    p := Parser{toks = toks[:]}
    rc := 0
    err_msg := ""
    for p.toks[p.i].kind != .EOF {
        st: Stmt
        if !parse(&p, &st) {
            return 1, "parse error"
        }
        if st.is_begin {
            db.in_txn = true
        } else if st.is_commit {
            db.in_txn = false
        } else if st.is_create {
            if !schema_add(&db.schema, st.ct_name, st.ct_cols) {
                err_msg = fmt.tprintf("table already exists: %s", st.ct_name)
                rc = 1
                break
            }
            append(&db.stores, RowStore{})
            continue
        } else if st.is_insert {
            idx := schema_index(&db.schema, st.ins_table)
            if idx < 0 {
                err_msg = fmt.tprintf("no such table: %s", st.ins_table)
                rc = 1
                break
            }
            row: Row
            for v in st.ins_vals {
                append(&row.vals, v_copy(v))
            }
            append(&db.stores[idx].rows, row)
            delete(st.ins_vals)
        } else if st.is_select {
            s := new(Stmt_Handle)
            s.db = db
            s.st = st
