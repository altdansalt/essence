  |> fn(t) { list.append(t, [Token(TkEof, "")]) }

fn do_lex(chars: List(String), acc: List(Token)) -> List(Token) {
  case chars {
    [] -> acc
    [c, ..rest] -> {
      case c {
        " " | "\t" | "\n" | "\r" -> do_lex(rest, acc)
        "-" -> {
          case rest {
            ["-", ..r2] -> do_lex(skip_to_newline(r2), acc)
            _ -> do_lex(rest, [Token(TkPunct, c), ..acc])
          }
        }

            let kind = case is_keyword(id) { True -> TkKw; False -> TkId }
            do_lex(rest2, [Token(kind, id), ..acc])
          } else if is_digit(c) {
            let #(num, rest2) = take_while(chars, is_digit)
            do_lex(rest2, [Token(TkInt, num), ..acc])

            do_lex(skip_quote(rest2), [Token(TkString, s), ..acc])

              "," -> do_lex(rest, [Token(TkComma, c), ..acc])
