  let n = String.length src in

  let is_digit c = c >= '0' && c <= '9' in

    let op = if p_peek_kw p "and" then "AND" else "OR" in
    p_advance p;
    EBinop (op, l, parse_expr p)
  end else l

  let l = parse_primary p in
  if p_tok p .k = TKOP then begin
    let op = (p_tok p).t in p_advance p;
    EBinop (op, l, parse_primary p)

  | TKID | TKKW ->

    Buffer.add_string buf t.t; p_advance p;

      p_advance p;

      Buffer.add_string buf (p_tok p).t;

  if not (p_accept_kind p TKLP) then raise Parse_error;
  let vals = ref [] in
  while p_tok p .k <> TKRP && p_tok p .k <> TKEOF do
    let t = p_tok p in
    let v =
      if t.k = TKINT then begin p_advance p; VInt (int_of_string t.t) end
      else if t.k = TKSTRING then begin p_advance p; VText t.t end
      else if p_accept_kw p "null" then VNull
      else raise Parse_error
    in
    vals := v :: !vals;
    ignore (p_accept_kind p TKComma)

Wait, there's a problem with the insert loop. In the C code:

```c
while (P->t[P->i].kind!=TK_RP && P->t[P->i].kind!=TK_EOF){
    ...
    if (!p_accept(P,TK_COMMA,NULL)) break;
}
```

So it breaks if there's no comma. In my OCaml code, I do `ignore (p_accept_kind p TKComma)` which doesn't break. But the while condition checks for TKRP, so if there's no comma, the next token should be TKRP and the while exits. But what if there's something else? The C code would break, but mine would continue and try to parse another value, potentially failing.

Actually, for well-formed SQL, after a value without a comma, the next token should be RP. So both approaches work for valid SQL. For invalid SQL, the C code breaks and then checks for RP (which would fail), while my code would try to parse another value and fail with Parse_error. Both result in an error, so it's fine.

But wait, there's another issue. In the C code, after the while loop, it does `if (!p_accept(P,TK_RP,NULL)) return -1;`. If the while broke because of no comma, the current token should be RP. If the while exited because the current token is RP, then p_accept(RP) succeeds. So both cases work.

In my code, the while exits when the current token is RP (or EOF). Then `p_accept_kind p TKRP` should succeed if it's RP. If it's EOF, it won't succeed, but that's an error case anyway.

Actually, there's a subtle issue. In the C code, the while loop condition checks for RP at the start. If the first token after `(` is RP (empty values list), the while doesn't execute, and then p_accept(RP) succeeds. In my code, same behavior. Good.

But what about the comma? In the C code, after parsing a value, it checks for comma. If no comma, it breaks. Then the while condition is re-checked. But since we broke, we don't re-check. We go to p_accept(RP).

In my code, after parsing a value, I do `ignore (p_accept_kind p TKComma)`. If there's no comma, this returns false but we continue the while loop. The while condition checks for RP. If the next token is RP, we exit. If not, we try to parse another value.

For valid SQL like `VALUES (1, 2, 3)`, after parsing 1, we accept comma, parse 2, accept comma, parse 3, no comma, while checks: current is RP, exit. Then accept RP. Works.

For `VALUES (1)`, after parsing 1, no comma, while checks: current is RP, exit. Then accept RP. Works.

So the behavior is equivalent for valid SQL. Good.

  if p_accept_kind p TKStar then sel_star := true
  else begin

    while !go do
      if p_accept_kind p TKStar then
        sel_cols := { col=None; star=true } :: !sel_cols
      else if p_tok p .k = TKID || p_tok p .k = TKKW then begin
        let buf = Buffer.create 16 in
        Buffer.add_string buf (p_tok p).t; p_advance p;
        if p_tok p .k = TKDot then begin
          p_advance p;
          if p_accept_kind p TKStar then
            sel_cols := { col=None; star=true } :: !sel_cols
          else begin
            Buffer.add_char buf '.';
            Buffer.add_string buf (p_tok p).t; p_advance p;
            sel_cols := { col=Some (Buffer.contents buf); star=false } :: !sel_cols
          end
        end else
          sel_cols := { col=Some (Buffer.contents buf); star=false } :: !sel_cols
      end else raise Parse_error;
      go := p_accept_kind p TKComma
    done
  end;
  if not (p_accept_kw p "from") then raise Parse_error;
  if p_tok p .k <> TKID then raise Parse_error;
  let sel_tables = ref [] in
  let parse_table_ref ?(check_inner=true) () =
    let tname = (p_tok p).t in p_advance p;
    let can_alias =
      p_tok p .k = TKID &&
      not (p_peek_kw p "on") && not (p_peek_kw p "where") && not (p_peek_kw p "order") &&
      (if check_inner then not (p_peek_kw p "inner") && not (p_peek_kw p "join") else true)
    in
    let alias = if can_alias then (let a = (p_tok p).t in p_advance p; Some a) else None in
    sel_tables := { tname; alias } :: !sel_tables
  in
  parse_table_ref ();
  let sel_join_on = ref None in
  let go = ref true in
  while !go do
    go := false;
    if p_peek_kw p "inner" || p_peek_kw p "join" then begin
      go := true;
      if p_accept_kw p "inner" then ignore (p_accept_kw p "join")
      else ignore (p_accept_kw p "join");
      if p_tok p .k <> TKID then raise Parse_error;
      parse_table_ref ~check_inner:false ();
      if p_accept_kw p "on" then begin
        let on = parse_expr p in
        sel_join_on := match !sel_join_on with
          | None -> Some on | Some prev -> Some (EBinop ("AND", prev, on))

    end
  done;
  let sel_where = if p_accept_kw p "where" then Some (parse_expr p) else None in
  let sel_order_col = ref None in
  let sel_order_desc = ref false in
  if p_accept_kw p "order" then begin

    let t = p_tok p in
    if t.k <> TKID && t.k <> TKKW then raise Parse_error;

(* Row context for expression evaluation *)
type row_ctx = {
  refs: table_ref list;
  tabs: table list;
  rows: row array;
}

let rowctx_colindex ctx col =
  match String.index_opt col '.' with
  | Some di ->
    let tname = String.sub col 0 di in
    let cname = String.sub col (di+1) (String.length col - di - 1) in
    let lt = String.lowercase_ascii tname in
    let rec aux i refs tabs =
      match refs, tabs with
      | r :: rs, t :: ts ->
        let tn = match r.alias with Some a -> a | None -> r.tname in
        if String.lowercase_ascii tn = lt then
          let ci = colindex t cname in
          if ci >= 0 then Some (i, ci) else None
        else aux (i+1) rs ts
      | _ -> None
    in aux 0 ctx.refs ctx.tabs
  | None ->
    let rec aux i refs tabs =
      match refs, tabs with
      | _, t :: ts ->
        let ci = colindex t col in
        if ci >= 0 then Some (i, ci) else aux (i+1) refs ts
      | _ -> None
    in aux 0 ctx.refs ctx.tabs

let rec eval_expr e ctx =
  match e with
  | EInt i -> VInt i
  | EStr s -> VText s
  | ECol col ->
    (match rowctx_colindex ctx col with
     | Some (ti, ci) -> v_copy ctx.rows.(ti).(ci)
     | None -> VNull)
  | EBinop (op, l, r) ->
    let a = eval_expr l ctx in
    let b = eval_expr r ctx in
    let op_u = String.uppercase_ascii op in
    if op_u = "AND" then VInt (if v_truthy a && v_truthy b then 1 else 0)
    else if op_u = "OR" then VInt (if v_truthy a || v_truthy b then 1 else 0)
    else begin
      let c = v_cmp a b in
      VInt (match op with
        | "=" | "==" -> if c=0 then 1 else 0
        | "<>" | "!=" -> if c<>0 then 1 else 0
        | "<" -> if c<0 then 1 else 0
        | ">" -> if c>0 then 1 else 0
        | "<=" -> if c<=0 then 1 else 0
        | ">=" -> if c>=0 then 1 else 0
        | _ -> 0)
    end

let row_match where ctx =
  match where with None -> true | Some e -> v_truthy (eval_expr e ctx)
