```vim9script
  def Accept(kind: number, text: string = ''): bool

    endif
    if text != '' && this.Text() !=? text

    endif
    this.i += 1
    return true

  def AcceptKW(kw: string): bool

  enddef

  def PeekKW(kw: string): bool
    return this.Kind() == 4 && this.Text() ==? kw
  enddef

  def ParsePrimary(): any
    var k = this.Kind()
    if k == 2  # TK_INT
      var e = Expr.new(1)
      e.ival = str2nr(this.Text())
      this.i += 1
      return e
    endif
    if k == 3  # TK_STRING
      var e = Expr.new(2)
      e.sval = this.Text()
      this.i += 1
      return e
    endif
    if k == 1 || k == 4  # TK_ID or TK_KW
      var e = Expr.new(0)
      var buf = this.Text()
      this.i += 1
      if this.Kind() == 12  # TK_DOT

        buf ..= '.' .. this.Text()
        this.i += 1
      endif
      e.col = buf
      return e
    endif
    if this.Accept(8)  # TK_LP
      var e = this.ParseExpr()
      this.Accept(9)  # TK_RP
      return e
    endif
    return null

  def ParseCmp(): any

      var e = Expr.new(3)
      e.op = this.Text()
      this.i += 1
      e.l = l
      e.r = this.ParsePrimary()
      return e
    endif
    return l
  enddef

  def ParseExpr(): any
```

1. Lex the SQL
2. Parse each statement
3. Execute each statement (CREATE, INSERT, SELECT)

```vim9script
def Exec(sql: string, cb: func(any, list<string>): void = null, arg: any = null): number
  var toks = Lex(sql)
  var p = Parser.new(toks)
  while p.Kind() != 0  # TK_EOF
    var st = p.Parse()
    if st == null || st.error
      this.errmsg = 'parse error'
      return 1
    endif
    if st.is_begin
      this.in_txn = true
    elseif st.is_commit
      this.in_txn = false
    elseif st.is_create
      var rc = this.ExecCreate(st)
      if rc != 0
        return rc
      endif
    elseif st.is_insert
      var rc = this.ExecInsert(st)
      if rc != 0
        return rc
      endif
    elseif st.is_select
      var stmt = Sqlite3Stmt.new(this, st)
      while stmt.Step() == 100
        if cb != null
          var vals: list<string> = []
          for i in range(stmt.nout)
            add(vals, stmt.outrow[i].ToText())
          endfor
          cb(arg, vals)
        endif
      endwhile
    endif
  endwhile
  return 0
enddef
```
