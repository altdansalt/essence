```vim
# Value helpers - using dict {type: 'null'|'int'|'text', ival: number, sval: string}
def VNull(): dict<any>
  return {type: 'null', ival: 0, sval: ''}
enddef

def VInt(i: number): dict<any>
  return {type: 'int', ival: i, sval: ''}
enddef

def VText(s: string): dict<any>
  return {type: 'text', ival: 0, sval: s}
enddef

def VCopy(v: dict<any>): dict<any>
  if v.type == 'text'
    return VText(v.sval)
  endif
  return copy(v)
enddef

def VCmp(a: dict<any>, b: dict<any>): number
  if a.type != b.type
    # order: null < int < text (matching C enum order)
    var at: number = a.type == 'null' ? 0 : (a.type == 'int' ? 1 : 2)
    var bt: number = b.type == 'null' ? 0 : (b.type == 'int' ? 1 : 2)
    return at - bt
  endif
  if a.type == 'int'
    return a.ival < b.ival ? -1 : (a.ival > b.ival ? 1 : 0)
  endif
  if a.type == 'text'
    return strcmp(a.sval, b.sval)
  endif
  return 0
enddef

def VToText(v: dict<any>): string
  if v.type == 'int'
    return string(v.ival)
  endif
  if v.type == 'text'
    return v.sval
  endif
  return 'NULL'
enddef

def VTruthy(v: dict<any>): bool
  return VCmp(v, VInt(0)) != 0
enddef
```

```vim
def Lex(sql: string): list<dict<string>>
  var toks: list<dict<string>> = []
  var pos = 0
  var n = len(sql)
  while pos < n
    var c = sql[pos]
    if c == ' ' || c == "\t" || c == "\n" || c == "\r"
      pos += 1
      continue
    endif
    if c == '-' && pos + 1 < n && sql[pos + 1] == '-'
      while pos < n && sql[pos] != "\n"
        pos += 1
      endwhile
      continue
    endif
    # identifier or keyword
    if c =~ '\a' || c == '_'
      var st = pos
      while pos < n && (sql[pos] =~ '\a' || sql[pos] =~ '\d' || sql[pos] == '_')
        pos += 1
      endwhile
      var word = sql[st : pos - 1]
      add(toks, {kind: IsKw(word) ? 'kw' : 'id', text: word})
      continue
    endif
    # number
    if c =~ '\d'
      var st = pos
      while pos < n && sql[pos] =~ '\d'
        pos += 1
      endwhile
      add(toks, {kind: 'int', text: sql[st : pos - 1]})
      continue
    endif
    # string literal
    if c == "'"
      pos += 1
      var st = pos
      while pos < n && sql[pos] != "'"
        pos += 1
      endwhile
      add(toks, {kind: 'string', text: pos > st ? sql[st : pos - 1] : ''})
      if pos < n
        pos += 1
      endif
      continue
    endif
    # quoted identifier
    if c == '"'
      pos += 1
      var st = pos
      while pos < n && sql[pos] != '"'
        pos += 1
      endwhile
      add(toks, {kind: 'id', text: pos > st ? sql[st : pos - 1] : ''})
      if pos < n
        pos += 1
      endif
      continue
    endif
    # operators and punctuation
    if c == '*'
      add(toks, {kind: 'star', text: '*'})
      pos += 1
    elseif c == ','
      add(toks, {kind: 'comma', text: ','})
      pos += 1
    elseif c == '('
      add(toks, {kind: 'lp', text: '('})
      pos += 1
    elseif c == ')'
      add(toks, {kind: 'rp', text: ')'})
      pos += 1
    elseif c == ';'
      add(toks, {kind: 'semi', text: ';'})
      pos += 1
    elseif c == '.'
      add(toks, {kind: 'dot', text: '.'})
      pos += 1
    elseif c == '=' || c == '<' || c == '>' || c == '!'
      var st = pos
      if pos + 1 < n && sql[pos + 1] == '='
        pos += 1
      endif
      if c == '<' && pos + 1 < n && sql[pos + 1] == '>'
        pos += 1
      endif
      if c == '!' && pos + 1 < n && sql[pos + 1] == '='
        pos += 1
      endif
      pos += 1
      add(toks, {kind: 'op', text: sql[st : pos - 1]})
    else
      add(toks, {kind: 'punct', text: c})
      pos += 1
    endif
  endwhile
  add(toks, {kind: 'eof', text: ''})
  return toks
enddef
```

```vim
class Parser
  var toks: list<dict<string>>
  var i: number = 0

  def new(toks: list<dict<string>>)
    this.toks = toks
  enddef

  def Peek(): dict<string>
    return this.toks[this.i]
  enddef

  def Accept(kind: string, text: string = ''): bool
    var t = this.toks[this.i]
    if t.kind != kind
      return false
    endif
    if text != '' && t.text !=? text
      return false
    endif
    this.i += 1
    return true
  enddef

  def AcceptKw(kw: string): bool
    return this.Accept('kw', kw)
  enddef

  def PeekKw(kw: string): bool
    var t = this.toks[this.i]
    return t.kind == 'kw' && t.text ==? kw
  enddef
```

```vim
  def ParsePrimary(): dict<any>
    var t = this.Peek()
    if t.kind == 'int'
      this.i += 1
      return {kind: 'int', ival: str2nr(t.text), sval: '', op: '', col: '', l: {}, r: {}}
    endif
    if t.kind == 'string'
      this.i += 1
      return {kind: 'str', ival: 0, sval: t.text, op: '', col: '', l: {}, r: {}}
    endif
    if t.kind == 'id' || t.kind == 'kw'
      var col = t.text
      this.i += 1
      if this.Peek().kind == 'dot'
        this.i += 1
        col ..= '.' .. this.Peek().text
        this.i += 1
      endif
      return {kind: 'col', ival: 0, sval: '', op: '', col: col, l: {}, r: {}}
    endif
    if this.Accept('lp')
      var e = this.ParseExpr()
      this.Accept('rp')
      return e
    endif
    return {}
  enddef

  def ParseCmp(): dict<any>
    var l = this.ParsePrimary()
    if this.Peek().kind == 'op'
      var op = this.Peek().text
      this.i += 1
      var r = this.ParsePrimary()
      return {kind: 'binop', ival: 0, sval: '', op: op, col: '', l: l, r: r}
    endif
    return l
  enddef

  def ParseExpr(): dict<any>
    var l = this.ParseCmp()
    if this.PeekKw('and') || this.PeekKw('or')
      var op = this.PeekKw('and') ? 'AND' : 'OR'
      this.i += 1
      var r = this.ParseExpr()
      return {kind: 'binop', ival: 0, sval: '', op: op, col: '', l: l, r: r}
    endif
    return l
  enddef
```

```vim
    return st

  def ParseInsert(): dict<any>
    var st: dict<any> = {is_insert: true, ins_table: '', ins_vals: []}
    if !this.AcceptKw('into')
      return {}
    endif
    if this.Peek().kind != 'id'
      return {}
    endif
    st.ins_table = this.Peek().text
    this.i += 1
    if !this.AcceptKw('values')
      return {}
    endif
    if !this.Accept('lp')
      return {}
    endif
    while this.Peek().kind != 'rp' && this.Peek().kind != 'eof'
      var t = this.Peek()
      if t.kind == 'int'
        add(st.ins_vals, VInt(str2nr(t.text)))
        this.i += 1
      elseif t.kind == 'string'
        add(st.ins_vals, VText(t.text))
        this.i += 1
      elseif this.AcceptKw('null')
        add(st.ins_vals, VNull())
      else
        return {}
      endif
      if !this.Accept('comma')
        break
      endif
    endwhile
    if !this.Accept('rp')
      return {}
    endif
    return st
  enddef

  def ParseSelect(): dict<any>
    var st: dict<any> = {is_select: true, sel_cols: [], sel_star: false, sel_tables: [], sel_where: {}, sel_order_col: '', sel_order_desc: false, sel_join_on: {}}
    if this.Accept('star')
      st.sel_star = true
    else
      while true
        var rc: dict<any> = {col: '', star: false}
        if this.Accept('star')
          rc.star = true
        elseif this.Peek().kind == 'id' || this.Peek().kind == 'kw'
          var col = this.Peek().text
          this.i += 1
          if this.Peek().kind == 'dot'
            this.i += 1
            if this.Accept('star')
              rc.star = true
            else
              col ..= '.' .. this.Peek().text
              this.i += 1
              rc.col = col
            endif
          else
            rc.col = col
          endif
        else
          return {}
        endif
        add(st.sel_cols, rc)
        if !this.Accept('comma')
          break
        endif
      endwhile
    endif
    if !this.AcceptKw('from')
      return {}
    endif
    if this.Peek().kind != 'id'
      return {}
    endif
    # first table
    var tr: dict<any> = {tname: this.Peek().text, alias: ''}
    this.i += 1
    if this.Peek().kind == 'id' && !this.PeekKw('on') && !this.PeekKw('where') && !this.PeekKw('order') && !this.PeekKw('inner') && !this.PeekKw('join')
      tr.alias = this.Peek().text
      this.i += 1
    endif
    add(st.sel_tables, tr)
    # joins
    while this.PeekKw('inner') || this.PeekKw('join')
      if this.AcceptKw('inner')
        if !this.AcceptKw('join')
          return {}
        endif
      else
        this.AcceptKw('join')
      endif
      if this.Peek().kind != 'id'
        return {}
      endif
      var tr2: dict<any> = {tname: this.Peek().text, alias: ''}
      this.i += 1
      if this.Peek().kind == 'id' && !this.PeekKw('on') && !this.PeekKw('where') && !this.PeekKw('order')

        this.i += 1
      endif
      add(st.sel_tables, tr2)
      if this.AcceptKw('on')
        var on = this.ParseExpr()
        if empty(st.sel_join_on)
          st.sel_join_on = on
        else
          st.sel_join_on = {kind: 'binop', op: 'AND', l: st.sel_join_on, r: on, ival: 0, sval: '', col: ''}
        endif
      endif
    endwhile
    if this.AcceptKw('where')
      st.sel_where = this.ParseExpr()
    endif
    if this.AcceptKw('order')
      if !this.AcceptKw('by')
        return {}
      endif
      var t = this.Peek()
      if t.kind != 'id' && t.kind != 'kw'
        return {}
      endif
      st.sel_order_col = t.text
      this.i += 1
      if this.AcceptKw('desc')
        st.sel_order_desc = true
      else
        this.AcceptKw('asc')
      endif
    endif
    return st
  enddef

  def Parse(): dict<any>
    if this.PeekKw('begin')
      this.i += 1
      this.AcceptKw('transaction')
      return {is_begin: true}
    endif
    if this.PeekKw('commit')
      this.i += 1
      this.AcceptKw('transaction')
      return {is_commit: true}

        return this.ParseCreate()
```

```vim
class Sqlite3
  var tables: list<string> = []        # table names
  var cols: list<list<string>> = []    # column names per table
  var stores: list<list<list<dict<any>>>> = []  # rows per table, each row is list of Values
  var errmsg: string = ''
  var in_txn: bool = false

  def new()
  enddef

  def FindTable(name: string): number
    for i in range(len(this.tables))
      if this.tables[i] ==? name
        return i
      endif
    endfor
    return -1
  enddef

  def ColIndex(tabidx: number, col: string): number
    for i in range(len(this.cols[tabidx]))
      if this.cols[tabidx][i] ==? col
        return i
      endif
    endfor
    return -1
  enddef
```

```vim
class Sqlite3Stmt
  var db: Sqlite3
  var st: dict<any>
  var sorted: list<list<dict<any>>> = []  # materialized rows
  var cur: number = 0
  var started: bool = false
  var outrow: list<dict<any>> = []
  var nout: number = 0

  def new(db: Sqlite3, st: dict<any>)
    this.db = db
    this.st = st
  enddef

  def ColIndexInCtx(col: string, tabs: list<number>, rows: list<list<dict<any>>>): list<number>
    # returns [tabidx, colidx] or [-1, -1]
    var dot = stridx(col, '.')
    if dot >= 0
      var tname = col[: dot - 1]
      var cname = col[dot + 1 :]
      for i in range(len(this.st.sel_tables))
        var tn = !empty(this.st.sel_tables[i].alias) ? this.st.sel_tables[i].alias : this.st.sel_tables[i].tname
        if tn ==? tname
          var ci = this.db.ColIndex(tabs[i], cname)
          if ci >= 0
            return [i, ci]
          endif
        endif
      endfor
      return [-1, -1]
    endif
    for i in range(len(tabs))
      var ci = this.db.ColIndex(tabs[i], col)
      if ci >= 0
        return [i, ci]
      endif
    endfor
    return [-1, -1]
  enddef

  def EvalExpr(e: dict<any>, tabs: list<number>, rows: list<list<dict<any>>>): dict<any>
    if empty(e)
      return VNull()
    endif
    if e.kind == 'int'
      return VInt(e.ival)
    endif
    if e.kind == 'str'
      return VText(e.sval)
    endif
    if e.kind == 'col'
      var res = this.ColIndexInCtx(e.col, tabs, rows)
      if res[0] < 0
        return VNull()
      endif
      return VCopy(rows[res[0]][res[1]])
    endif
    if e.kind == 'binop'
      var a = this.EvalExpr(e.l, tabs, rows)
      var b = this.EvalExpr(e.r, tabs, rows)
      if e.op ==? 'AND'
        return VInt(VTruthy(a) && VTruthy(b) ? 1 : 0)
      elseif e.op ==? 'OR'
        return VInt(VTruthy(a) || VTruthy(b) ? 1 : 0)
      else
        var cmp = VCmp(a, b)
        if e.op == '='
          return VInt(cmp == 0 ? 1 : 0)
        elseif e.op == '<>'
          return VInt(cmp != 0 ? 1 : 0)
        elseif e.op == '<'
          return VInt(cmp < 0 ? 1 : 0)
        elseif e.op == '>'
          return VInt(cmp > 0 ? 1 : 0)
        elseif e.op == '<='
          return VInt(cmp <= 0 ? 1 : 0)
        elseif e.op == '>='
          return VInt(cmp >= 0 ? 1 : 0)
        endif
      endif
    endif
    return VNull()
  enddef

  def RowMatch(where: dict<any>, tabs: list<number>, rows: list<list<dict<any>>>): bool
    if empty(where)
      return true
    endif
    return VTruthy(this.EvalExpr(where, tabs, rows))
  enddef

  def CollectMatching()
    var n = len(this.st.sel_tables)
    var idxs: list<number> = []
    for i in range(n)
      add(idxs, this.db.FindTable(this.st.sel_tables[i].tname))
    endfor
    # nested loop
    var curs: list<number> = []  # current row index in each table
    for i in range(n)
      add(curs, 0)
    endfor
    while true
      # check all cursors valid
      var ok = true
      for i in range(n)
        if idxs[i] < 0 || curs[i] >= len(this.db.stores[idxs[i]])
          ok = false
          break
        endif
      endfor
      if !ok
        break
      endif
      # build row context
      var tabs = idxs
      var rows: list<list<dict<any>>> = []
      for i in range(n)
        add(rows, this.db.stores[idxs[i]][curs[i]])
      endfor
      if this.RowMatch(this.st.sel_where, tabs, rows) && (empty(this.st.sel_join_on) || this.RowMatch(this.st.sel_join_on, tabs, rows))
        # collect combined row
        var combined: list<dict<any>> = []
        for i in range(n)
          for j in range(len(rows[i]))
            add(combined, VCopy(rows[i][j]))
          endfor
        endfor
        add(this.sorted, combined)
      endif
      # advance innermost
      var k = n - 1
      while k >= 0
        curs[k] += 1
        if idxs[k] >= 0 && curs[k] < len(this.db.stores[idxs[k]])
          break
        endif
        if k > 0
          curs[k] = 0
        endif
        k -= 1
      endwhile
      if k < 0
        break
      endif
    endwhile
    # ORDER BY
    if !empty(this.st.sel_order_col) && n == 1
      var cidx = this.db.ColIndex(idxs[0], this.st.sel_order_col)
      if cidx >= 0
        # insertion sort
        for i in range(1, len(this.sorted) - 1)
          var j = i
          while j > 0
            var a = this.sorted[j - 1][cidx]
            var b = this.sorted[j][cidx]
            var cmp = VCmp(a, b)
            if (this.st.sel_order_desc && cmp < 0) || (!this.st.sel_order_desc && cmp > 0)
              var tmp = this.sorted[j - 1]
              this.sorted[j - 1] = this.sorted[j]
              this.sorted[j] = tmp
            else
              break
            endif
            j -= 1
          endwhile
        endfor
      endif
    endif
  enddef

  def Step(): number
    if !this.started
      this.started = true
      this.CollectMatching()
    endif
    if this.cur >= len(this.sorted)
      return 101  # DONE
    endif
```

```vim
def Lex(sql: string): list<dict<string>>
  var toks: list<dict<string>> = []
  var pos = 0
  var n = len(sql)
  while pos < n
    var c = sql[pos]
    if c == ' ' || c == "\t" || c == "\n" || c == "\r"
      pos += 1
      continue
    endif
    if c == '-' && pos + 1 < n && sql[pos + 1] == '-'
      while pos < n && sql[pos] != "\n"
        pos += 1
      endwhile
      continue
    endif
    if c =~ '\a' || c == '_'
      var st = pos
      while pos < n && (sql[pos] =~ '\a' || sql[pos] =~ '\d' || sql[pos] == '_')
        pos += 1
      endwhile
      var w = sql[st : pos - 1]
      add(toks, {kind: IsKw(w) ? 'kw' : 'id', text: w})
      continue
    endif
    if c =~ '\d'
      var st = pos
      while pos < n && sql[pos] =~ '\d'
        pos += 1
      endwhile
      add(toks, {kind: 'int', text: sql[st : pos - 1]})
      continue
    endif
    if c == "'"
      pos += 1
      var st = pos
      while pos < n && sql[pos] != "'"
        pos += 1
      endwhile
      add(toks, {kind: 'string', text: sql[st : pos - 1]})
      if pos < n
        pos += 1
      endif
      continue
    endif
    if c == '"'
      pos += 1
      var st = pos
      while pos < n && sql[pos] != '"'
        pos += 1
      endwhile
      add(toks, {kind: 'id', text: sql[st : pos - 1]})
      if pos < n
        pos += 1
      endif
      continue
    endif
    if c == '*'
      add(toks, {kind: 'star', text: '*'})
      pos += 1
    elseif c == ','
      add(toks, {kind: 'comma', text: ','})
      pos += 1
    elseif c == '('
      add(toks, {kind: 'lp', text: '('})
      pos += 1
    elseif c == ')'
      add(toks, {kind: 'rp', text: ')'})
      pos += 1
    elseif c == ';'
      add(toks, {kind: 'semi', text: ';'})
      pos += 1
    elseif c == '.'
      add(toks, {kind: 'dot', text: '.'})
      pos += 1
    elseif c == '=' || c == '<' || c == '>' || c == '!'
      var st = pos
      if pos + 1 < n && sql[pos + 1] == '='
        pos += 1
      endif
      if c == '<' && pos + 1 < n && sql[pos + 1] == '>'
        pos += 1
      endif
      if c == '!' && pos + 1 < n && sql[pos + 1] == '='
        pos += 1
      endif
      pos += 1
      add(toks, {kind: 'op', text: sql[st : pos - 1]})
    else
      add(toks, {kind: 'punct', text: c})
      pos += 1
    endif
  endwhile
  add(toks, {kind: 'eof', text: ''})
  return toks
enddef
```

class Parser
  var toks: list<dict<string>>
  var i: number = 0

  def new(toks: list<dict<string>>)
    this.toks = toks
  enddef

  def Peek(): dict<string>
    return this.toks[this.i]
  enddef

  def Accept(kind: string, text: string = ''): bool
    var t = this.toks[this.i]
    if t.kind != kind
      return false
    endif
    if text != '' && t.text !=? text
      return false
    endif
    this.i += 1
    return true
  enddef

  def AcceptKw(kw: string): bool
    return this.Accept('kw', kw)
  enddef

  def PeekKw(kw: string): bool
    return this.toks[this.i].kind == 'kw' && this.toks[this.i].text ==? kw
  enddef

  def ParsePrimary(): dict<any>
    var t = this.Peek()
    if t.kind == 'int'
      this.i += 1
      return {kind: 'int', ival: str2nr(t.text), sval: '', op: '', col: '', l: {}, r: {}}
    endif
    if t.kind == 'string'
      this.i += 1
      return {kind: 'str', ival: 0, sval: t.text, op: '', col: '', l: {}, r: {}}
    endif
    if t.kind == 'id' || t.kind == 'kw'
      var col = t.text
      this.i += 1
      if this.Peek().kind == 'dot'
        this.i += 1
        col ..= '.' .. this.Peek().text
        this.i += 1
      endif
      return {kind: 'col', ival: 0, sval: '', op: '', col: col, l: {}, r: {}}
    endif
    if this.Accept('lp')
      var e = this.ParseExpr()
      this.Accept('rp')
      return e
    endif
    return {}
  enddef

  def ParseCmp(): dict<any>
    var l = this.ParsePrimary()
    if this.Peek().kind == 'op'
      var op = this.Peek().text
      this.i += 1
      var r = this.ParsePrimary()
      return {kind: 'binop', ival: 0, sval: '', op: op, col: '', l: l, r: r}
    endif
    return l
  enddef

  def ParseExpr(): dict<any>
    var l = this.ParseCmp()
    if this.PeekKw('and') || this.PeekKw('or')
      var op = this.PeekKw('and') ? 'AND' : 'OR'
      this.i += 1
      var r = this.ParseExpr()
      return {kind: 'binop', ival: 0, sval: '', op: op, col: '', l: l, r: r}
    endif
    return l
  enddef

    if t.kind != 'id' && t.kind != 'kw'
      return {}
    endif

    this.i += 1
    if !this.Accept('lp')
      return {}
