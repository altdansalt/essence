```nostos
exec(db, sql, cb) = {
    toks = lex(sql)
    p = PState(toks, 0)
    var rc = 0
    while p.toks[p.i].kind != TEOF && rc == 0 {
        try {
            ast = parse_one(p)
            p_accept_kind(p, TSemi)
            match ast {
                SBegin -> { db.in_txn = true }
                SCommit -> { db.in_txn = false }
                SCreate(name, cols) -> {
                    if schema_find(db, name) >= 0 then {
                        db.errmsg = "table already exists: " ++ name
                        rc = 1
                    } else {
                        db.tables = db.tables ++ [(name, cols)]
                        db.stores = db.stores ++ [[]]
                    }
                }
                SInsert(table, vals) -> {
                    idx = schema_find(db, table)
                    if idx < 0 then {
                        db.errmsg = "no such table: " ++ table
                        rc = 1
                    } else {
                        db.stores = update_at(db.stores, idx, 
                            (rows) => rows ++ [vals])
                    }
                }
                SSelect(sel) -> {
                    # Execute select and call callback for each row
                    rows = collect_rows(db, sel)
                    tabs = sel.tables.map((t) => {
                        i = schema_find(db, t.tname)
                        if i >= 0 then db.tables[i].1 else []
                    })
                    offsets = compute_offsets(tabs)
                    rows.forEach((row) => {
                        outrow = project_row(sel, tabs, offsets, row)
                        cb(outrow.length(), outrow.map(v_to_text))
                    })
                }
            }
        } catch {
            e -> {
                db.errmsg = "parse error"
                rc = 1
            }
        }
    }
    rc
}
```
